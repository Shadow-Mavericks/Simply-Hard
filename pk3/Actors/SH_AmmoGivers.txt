// Must match values in STRIKERS_CODE.ACS.
const int AMMO_CLIP			= 0;
const int AMMO_CLIPBOX		= 1;
const int AMMO_SHELLS		= 2;
const int AMMO_SHELLBOX		= 3;
const int AMMO_ROCKET		= 4;
const int AMMO_ROCKETBOX	= 5;
const int AMMO_CELL			= 6;
const int AMMO_CELLBOX		= 7;

actor ClipGiver : AmmoGiver_Base replaces Clip
{
	States
	{
		Base:
			CLIP A 0 A_SetScale(1.0)
			goto Idle
		
		MaximusPrime:
			ENGU A 0 A_SetScale(0.8)
			goto Idle
			
		Pickup:
			"####" "#" 0 A_JumpIf(CallACS("AMMO_PICKUP", AMMO_CLIP, false) == 0, "Failure")
			"####" "#" 0 A_RailWait
			stop

		Dropped:
			TNT1 A 0 A_SpawnItemEx("ClipGiver_Dropped", 0, 0, 0, velx, vely, velz, angle, SXF_ABSOLUTEVELOCITY | SXF_ABSOLUTEANGLE, 0)
			stop
	}
}

actor ClipGiver_Dropped : ClipGiver
{
	States
	{

		Pickup:
