actor ClipGiver : AmmoGiver_Base replaces Clip
{
	States
	{
		Base:
			CLIP A 70
			goto CheckSprite
		
		MaximusPrime:
			ENGU A 70
			goto CheckSprite
			
		Gosimer:
			D3DA A 70
			goto CheckSprite

		Init:
			TNT1 A 0 A_SetArg(1, AMMO_CLIP) // Ammo Type
			goto CheckSprite

		SetScale:
			"####" "#" 0 A_SetUserVar(user_scale_Gosimer, 30)
			"####" "#" 0 A_SetUserVar(user_scale_MaximusPrime, 80)
			goto Idle
	}
}
