actor ShotgunGiver : WeaponGiver_Base replaces Shotgun
{
	States
	{
		Base:
			SHOT A 70
			goto CheckSprite
			
		Kunai:
			SNGL Z 70
			goto CheckSprite

		Gosimer:
			DSHT A 70
			goto CheckSprite
		
		MaximusPrime:
			AGUN A 70
			goto CheckSprite

		MaximusPrimeDropped:
			BRIF A 70
			goto CheckSprite

		Init:
			TNT1 A 0 A_SetArg(1, WEAPON_SHOTGUN) // Ammo Type
			goto CheckSprite

		SetScale:
			"####" "#" 0 A_SetUserVar(user_scale_Gosimer, 50)
			"####" "#" 0 A_SetUserVar(user_scale_MaximusPrime, 80)
			goto Idle
	}
}
