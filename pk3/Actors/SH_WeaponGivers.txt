actor ChainsawGiver : WeaponGiver_Base
{
	//$Sprite 2SAWA0
	States
	{
		Base:
			2SAW A -1
			loop
			
		Kunai:
			KBLD Z -1
			loop

		Gosimer:
			2SAW A -1
			loop
		
		MaximusPrime:
			LGSB R -1
			loop

		Init:
			TNT1 A 0 A_SetArg(1, WEAPON_CHAINSAW) // Weapon Type
			goto CheckSprite

		SetScale:
			"####" "#" 0 A_SetUserVar(user_scale_MaximusPrime, 80)
			goto Idle
	}
}
